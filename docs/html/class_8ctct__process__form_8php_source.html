43  /**
44  * Process the form if there is a $_POST['uniqueformid'] set
45  *
46  * 1. Validates the data
47  * 2. Creates a KWSContact contact object
48  * 3. Validate the email based on settings
49  * 4. If valid, add/update
50  *
51  * @uses KWSConstantContact::addUpdateContact() To add/update contact
52  * @todo Return contact on success.
53  * @return WP_Error|KWSContact Returns a WP error if error, otherwise a contact object.
54  */
55  function process() {
56 
57  // Check that the form was submitted and we have an email value, otherwise return false
58  if(!isset($_POST['uniqueformid'])) { return false; }
59 
60  $this->id = esc_attr($_POST['uniqueformid']);
61 
62  // Validate the POST data
63  $data = $this->sanitizePost();
64 
65  // Create the contact
66  $KWSContact = new KWSContact($data);
67 
68  // Check If Email Is Real
69  $email_validation = $this->validateEmail($KWSContact);
70 
71  // Add the data to the object
72  $this->setResults('email_validation', $email_validation);
73 
74  #r($_POST);
75  #r($data);
76  #r($this);
77  #r($KWSContact, true);
78 
79  // If validation failed, stop processing
80  if(is_wp_error($email_validation)) { return $email_validation; }
81 
82  // Otherwise, let's Add/Update
83  return KWSConstantContact::getInstance()->addUpdateContact($KWSContact);
84  }

110  /**
111  * Sanitize the post data array before working with it
112  * @return array Sanitized $_POST array
113  */
114  private function sanitizePost() {
115 
116  unset($_POST['fields']['lists']);
117 
118  foreach($_POST['fields'] as $key => $value) {
119  if(isset($value['value'])) {
120  $output[$key] = esc_attr($value['value']);
121  } else {
122  $output[$key] = esc_attr($value);
123  }
124  }
125 
126  // Make sure lists are IDs
127  if(!empty($_POST['lists'])) {
128  foreach($_POST['lists'] as $list) {
129  if(!is_numeric($list)) { continue; }
130  // Constant Contact requires list IDs to be strings of numbers...
131  $output['lists'][] = (string)intval($list);
132  }
133  }
134 
135  unset($output['constant-contact-signup-submit']);
136 
137  return $output;
138  }

140  /**
141  * Validate email based on user settings.
142  *
143  * First, verify email using `is_email()` WordPress function (required)
144  *
145  * Then, process email validation based on settings.
146  *
147  * @param KWSContact $Contact Contact object
148  * @return WP_Error|boolean If valid, return `true`, otherwise return a WP_Error object.
149  */
150  function validateEmail(KWSContact $Contact) {
151 
152  $email = $Contact->get('email');
153 
154  $is_valid = array();
155 
156  // 1: Check if it's an email at all
157  if(empty($email)) {
158  do_action('ctct_debug', 'Empty email address', $email);
159  return new WP_Error('empty', __('Email address not defined.', 'constant-contact-api'));
160  }
161  if(!is_email($email)) {
162  do_action('ctct_debug', 'Invalid email address', $email);
163  return new WP_Error('not_email', __('Invalid email address.', 'constant-contact-api'));
164  }
165 
166  $methods = (array)CTCT_Settings::get('spam_methods');
167 
168  // 2: Akismet validation
169  if(in_array('akismet', $methods)) {
170  $akismetCheck = $this->akismetCheck($Contact);
171  if(is_wp_error($akismetCheck)) {
172  return $akismetCheck;
173  }
174  }
175 
176  // 3: WangGuard validation
177  if(in_array('wangguard', $methods) && function_exists('wangguard_verify_email') && wangguard_server_connectivity_ok()) {
178  global $wangguard_api_host;
179 
180  // If WangGuard isn't set up yet, set'er up!
181  if(empty($wangguard_api_host)) { wangguard_init(); }
182 
183  $return = wangguard_verify_email($email , wangguard_getRemoteIP() , wangguard_getRemoteProxyIP());
184 
185  if($return == 'checked' || $return == 'not-checked') {
186  do_action('ctct_debug', 'DataValidation validation passed.', $email, $return);
187  } else {
188  return new WP_Error('wangguard', 'Email validation failed.', $email, $return);
189  }
190  }
191 
192  // 4: DataValidation.com validation
193  if(in_array('datavalidation', $methods) && class_exists('DataValidation')) {
194  $Validate = new DataValidation(CTCT_Settings::get('datavalidation_api_key'));
195  $validation = $Validate->validate($email);
196 
197  if($validation === false) {
198  do_action('ctct_debug', 'DataValidation validation failed.', $email, $Validate);
199  $message = isset($Validate->message) ? $Validate->message : __('Not a valid email.', 'constant-contact-api');
200  return new WP_Error('datavalidation', $message, $email, $Validate);
201  } elseif($validation === null) {
202  do_action('ctct_debug', 'DataValidation validation inconclusive.', $email, $Validate);
203  } elseif($validation === true) {
204  do_action('ctct_debug', 'DataValidation validation passed.', $email, $Validate);
205  }
206  }
207 
208  // 5: SMTP validation
209  if(in_array('smtp', $methods) && class_exists('SMTP_validateEmail')) {
210 
211  $SMTP_Validator = new SMTP_validateEmail();
212 
213  $results = $SMTP_Validator->validate(array($email), get_option( 'admin_email' ));
214 
215  if($results[$email]) {
216  do_action('ctct_debug', 'SMTP validation passed.', $email, $results);
217  } else {
218  do_action('ctct_debug', 'SMTP validation failed.', $email, $results);
219  return new WP_Error('smtp', 'Email validation failed.', $email, $results);
220  }
221  }
222 
223  return true;
224  }

226  /**
227  * Verify the $_POST using Akismet
228  *
229  * @filter akismet_comment_nonce
230  * @link http://akismet.com/development/api/#comment-check
231  * @uses akismet_init()
232  * @uses akismet_auto_check_comment()
233  *
234  */
235  function akismetCheck() {
236  global $akismet_api_host, $akismet_api_port;
237 
238  if(!function_exists('akismet_http_post') || apply_filters('disable_constant_contact_akismet', false)) {
239  return true;
240  }
241 
242  /** If Akismet hasn't been initialized, initialize it. */
243  if(empty($akismet_api_port)) { akismet_init(); }
244 
245  /** Disable nonce verification */
246  add_filter('akismet_comment_nonce', function() { return 'inactive'; });
247 
248  ob_start();
249  $check = akismet_auto_check_comment(array());
250  ob_clean();
251 
252  if(empty($check) || empty($check['akismet_result'])) {
253  do_action('ctct_debug', 'There was an issue with Akismet: the response was invalid.', $check);
254  } else {
255  switch($check['akismet_result']) {
256  case 'true':
257  do_action('ctct_debug', __('Akismet caught this comment as spam'), $check);
258  return new WP_Error('akismet', __('Akismet caught this comment as spam'), $check);
259  break;
260  case 'false':
261  do_action('ctct_debug', __('Akismet cleared this comment'), $check);
262  break;
263  default:
264  do_action('ctct_debug', sprintf(__('Akismet was unable to check this comment (response: %s), will automatically retry again later.'), substr($check['akismet_result'], 0, 50)), $check);
265  }
266  }
267 
268  return true;
269  }
