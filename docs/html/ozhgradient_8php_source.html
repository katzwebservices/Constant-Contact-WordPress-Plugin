63  // Constructor. Creates, fills and returns an image
64  function gd_gradient_fill($w,$h,$d,$s,$e,$step=0) {
65  $this->width = $w;
66  $this->height = $h;
67  $this->direction = $d;
68  $this->startcolor = $s;
69  $this->endcolor = $e;
70  $this->step = intval(abs($step));
71  $this->path = dirname(__FILE__)."/gradients/"; // button cache directory
72  $this->imgname = $this->startcolor .'-'.$this->endcolor.'-'. $this->width . 'x'.$this->height;
73  // Attempt to create a blank image in true colors, or a new palette based image if this fails
74  if (function_exists('imagecreatetruecolor')) {
75  $this->image = imagecreatetruecolor($this->width,$this->height);
76  } elseif (function_exists('imagecreate')) {
77  $this->image = imagecreate($this->width,$this->height);
78  } else {
79  die('Unable to create an image');
80  }
81 
82  // Fill it
83  $this->fill($this->image,$this->direction,$this->startcolor,$this->endcolor);
84 
85  // Show it
86  $this->display($this->image);
87 
88  // Return it
89  return $this->image;
90  }

93  function retrieve($type = 'png') {
94  if($type == 'vnd.wap.wbmp') { $ext = 'bmp';} else { $ext = $type; }
95  if($bytes = @filesize("{$this->path}{$this->imgname}.$ext")) { // send cached version
96  // seconds, minutes, hours, days
97  $expires = 60*60*24*365*5;
98  header("Pragma: public");
99  header("Cache-Control: maxage=".$expires);
100  header('Expires: ' . gmdate('D, d M Y H:i:s', time()+$expires) . ' GMT');
101  header("Content-type: image/$type");
102  header("Content-Length: $bytes");
103  readfile("{$this->path}{$this->imgname}.$ext");
104  return true;
105  }
106  return false;

145  // The main function that draws the gradient
146  function fill($im,$direction,$start,$end) {
147 
148  switch($direction) {
149  case 'horizontal':
150  $line_numbers = imagesx($im);
151  $line_width = imagesy($im);
152  list($r1,$g1,$b1) = $this->hex2rgb($start);
153  list($r2,$g2,$b2) = $this->hex2rgb($end);
154  break;
155  case 'vertical':
156  $line_numbers = imagesy($im);
157  $line_width = imagesx($im);
158  list($r1,$g1,$b1) = $this->hex2rgb($start);
159  list($r2,$g2,$b2) = $this->hex2rgb($end);
160  break;
161  case 'ellipse':
162  $width = imagesx($im);
163  $height = imagesy($im);
164  $rh=$height>$width?1:$width/$height;
165  $rw=$width>$height?1:$height/$width;
166  $line_numbers = min($width,$height);
167  $center_x = $width/2;
168  $center_y = $height/2;
169  list($r1,$g1,$b1) = $this->hex2rgb($end);
170  list($r2,$g2,$b2) = $this->hex2rgb($start);
171  imagefill($im, 0, 0, imagecolorallocate( $im, $r1, $g1, $b1 ));
172  break;
173  case 'ellipse2':
174  $width = imagesx($im);
175  $height = imagesy($im);
176  $rh=$height>$width?1:$width/$height;
177  $rw=$width>$height?1:$height/$width;
178  $line_numbers = sqrt(pow($width,2)+pow($height,2));
179  $center_x = $width/2;
180  $center_y = $height/2;
181  list($r1,$g1,$b1) = $this->hex2rgb($end);
182  list($r2,$g2,$b2) = $this->hex2rgb($start);
183  break;
184  case 'circle':
185  $width = imagesx($im);
186  $height = imagesy($im);
187  $line_numbers = sqrt(pow($width,2)+pow($height,2));
188  $center_x = $width/2;
189  $center_y = $height/2;
190  $rh = $rw = 1;
191  list($r1,$g1,$b1) = $this->hex2rgb($end);
192  list($r2,$g2,$b2) = $this->hex2rgb($start);
193  break;
194  case 'circle2':
195  $width = imagesx($im);
196  $height = imagesy($im);
197  $line_numbers = min($width,$height);
198  $center_x = $width/2;
199  $center_y = $height/2;
200  $rh = $rw = 1;
201  list($r1,$g1,$b1) = $this->hex2rgb($end);
202  list($r2,$g2,$b2) = $this->hex2rgb($start);
203  imagefill($im, 0, 0, imagecolorallocate( $im, $r1, $g1, $b1 ));
204  break;
205  case 'square':
206  case 'rectangle':
207  $width = imagesx($im);
208  $height = imagesy($im);
209  $line_numbers = max($width,$height)/2;
210  list($r1,$g1,$b1) = $this->hex2rgb($end);
211  list($r2,$g2,$b2) = $this->hex2rgb($start);
212  break;
213  case 'diamond':
214  list($r1,$g1,$b1) = $this->hex2rgb($end);
215  list($r2,$g2,$b2) = $this->hex2rgb($start);
216  $width = imagesx($im);
217  $height = imagesy($im);
218  $rh=$height>$width?1:$width/$height;
219  $rw=$width>$height?1:$height/$width;
220  $line_numbers = min($width,$height);
221  break;
222  default:
223  }
224 
225  for ( $i = 0; $i < $line_numbers; $i=$i+1+$this->step ) {
226  // old values :
227  if(empty($r)) { $r = ''; }
228  if(empty($g)) { $g = ''; }
229  if(empty($b)) { $b = ''; }
230  $old_r=$r;
231  $old_g=$g;
232  $old_b=$b;
233  // new values :
234  $r = ( $r2 - $r1 != 0 ) ? intval( $r1 + ( $r2 - $r1 ) * ( $i / $line_numbers ) ): $r1;
235  $g = ( $g2 - $g1 != 0 ) ? intval( $g1 + ( $g2 - $g1 ) * ( $i / $line_numbers ) ): $g1;
236  $b = ( $b2 - $b1 != 0 ) ? intval( $b1 + ( $b2 - $b1 ) * ( $i / $line_numbers ) ): $b1;
237  // if new values are really new ones, allocate a new color, otherwise reuse previous color.
238  // There's a "feature" in imagecolorallocate that makes this function
239  // always returns '-1' after 255 colors have been allocated in an image that was created with
240  // imagecreate (everything works fine with imagecreatetruecolor)
241  if ( "$old_r,$old_g,$old_b" != "$r,$g,$b")
242  $fill = imagecolorallocate( $im, $r, $g, $b );
243  switch($direction) {
244  case 'vertical':
245  imagefilledrectangle($im, 0, $i, $line_width, $i+$this->step, $fill);
246  break;
247  case 'horizontal':
248  imagefilledrectangle( $im, $i, 0, $i+$this->step, $line_width, $fill );
249  break;
250  case 'ellipse':
251  case 'ellipse2':
252  case 'circle':
253  case 'circle2':
254  imagefilledellipse ($im,$center_x, $center_y, ($line_numbers-$i)*$rh, ($line_numbers-$i)*$rw,$fill);
255  break;
256  case 'square':
257  case 'rectangle':
258  imagefilledrectangle ($im,$i*$width/$height,$i*$height/$width,$width-($i*$width/$height), $height-($i*$height/$width),$fill);
259  break;
260  case 'diamond':
261  imagefilledpolygon($im, array (
262  $width/2, $i*$rw-0.5*$height,
263  $i*$rh-0.5*$width, $height/2,
264  $width/2,1.5*$height-$i*$rw,
265  1.5*$width-$i*$rh, $height/2 ), 4, $fill);
266  break;
267  default:
268  }
269  }
270  }

272  // #ff00ff -> array(255,0,255) or #f0f -> array(255,0,255)
273  function hex2rgb($color) {
274  $color = str_replace('#','',$color);
275  $s = strlen($color) / 3;
276  $rgb[]=hexdec(str_repeat(substr($color,0,$s),2/$s));
277  $rgb[]=hexdec(str_repeat(substr($color,$s,$s),2/$s));
278  $rgb[]=hexdec(str_repeat(substr($color,2*$s,$s),2/$s));
279  return $rgb;
280  }
