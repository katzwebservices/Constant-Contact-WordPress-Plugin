78  static function handle_error($errno, $errstr, $errfile, $errline) {
79  if (!(error_reporting() & $errno)) {
80  // this error code is not included in error_reporting
81  return;
82  }
83 
84  switch ($errno) {
85  case E_NOTICE:
86  case E_USER_NOTICE:
87  $ex = new PhpNotice($errstr, $errno, $errfile, $errline);
88  break;
89 
90  case E_WARNING:
91  case E_USER_WARNING:
92  $ex = new PhpWarning($errstr, $errno, $errfile, $errline);
93  break;
94 
95  case E_STRICT:
96  $ex = new PhpStrict($errstr, $errno, $errfile, $errline);
97  break;
98 
99  case E_PARSE:
100  $ex = new PhpParse($errstr, $errno, $errfile, $errline);
101  break;
102 
103  default:
104  $ex = new PhpError($errstr, $errno, $errfile, $errline);
105  }
106 
107  self::handle_exception($ex, false);
108 
109  if (self::$previous_error_handler) {
110  call_user_func(self::$previous_error_handler, $errno, $errstr, $errfile, $errline);
111  }
112  }

114  /*

119  static function handle_exception($exception, $call_previous = true) {
120  self::$exceptions[] = $exception;
