37 /**
38  * Create an email campaign with the parameters provided
39  * @param array $params associative array of parameters to create a campaign from
40  */
41 function createCampaign(array $params)
42 {
43  $cc = new ConstantContact(APIKEY);
44  $campaign = new Campaign();
45  $campaign->name = $params['name'];
46  $campaign->subject = $params['subject'];
47  $campaign->from_name = $params['from_name'];
48  $campaign->from_email = $params['from_addr'];
49  $campaign->greeting_string = $params['greeting_string'];
50  $campaign->reply_to_email = $params['reply_to_addr'];
51  $campaign->text_content = $params['text_content'];
52  $campaign->email_content = $params['email_content'];
53  $campaign->email_content_format = $params['format'];
54 
55  // add the selected list or lists to the campaign
56  if (isset($params['lists'])) {
57  if (count($params['lists']) > 1) {
58  foreach ($params['lists'] as $list) {
59  $campaign->addList($list);
60  }
61  } else {
62  $campaign->addList($params['lists'][0]);
63  }
64  }
65 
66  return $cc->addEmailCampaign(ACCESS_TOKEN, $campaign);
67 }

69 /**
70  * Create a schedule for a campaign - this is time the campaign will be sent out
71  * @param $campaignId - Id of the campaign to be scheduled
72  * @param $time - ISO 8601 formatted timestamp of when the campaign should be sent
73  */
74 function createSchedule($campaignId, $time)
75 {
76  $cc = new ConstantContact(APIKEY);
77  $schedule = new Schedule();
78  $schedule->scheduled_date = $time;
79  return $cc->addEmailCampaignSchedule(ACCESS_TOKEN, $campaignId, $schedule);
80 }
