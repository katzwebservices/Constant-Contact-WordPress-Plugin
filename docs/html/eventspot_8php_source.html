159  /**
160  * Get the date of the latest registration activity for an event.
161  * @param Event $event The event object
162  * @return string The date of the last registration activity
163  */
164  function latest_registrant($event) {
165 
166  $_registrants = $this->old_api->getRegistrants($event);
167 
168  foreach($_registrants['registrants'] as $key => &$reg) {
169  $latest = 0;
170  if(isset($reg->registrationStatus) && strtolower($reg->registrationStatus) !== 'cancelled') {
171  $timestamp = strtotime($reg->registrationDate);
172  $reg->registrationTimestamp = $timestamp;
173  if($timestamp > $latest) { $latest = $timestamp; }
174  } else {
175  unset($_registrants['registrants'][$key]);
176  }
177  }
178  if(empty($timestamp)) {
179  return __('N/A','constant-contact-api');
180  } else {
181  return apply_filters('cc_event_registrationdate', $timestamp);
182  }
183  }
