668  function reportObjectMapper($xml_string) {
669  $xml = simplexml_load_string($xml_string);
670 
671 
672  $results = null;
673  $results = array();
674 
675  $report_root_parameters = array();
676  $report_aggregate_metrics = array();
677 
678  //Load root parameters
679 
680  $report_root_parameters['updated'] = strval($xml->updated);
681  $report_root_parameters['generator'] = strval($xml->generator);
682  $report_root_parameters['generatorVersion'] = strval($xml->generator->attributes());
683 
684  $open_search_results = $xml->children('http://a9.com/-/spec/opensearchrss/1.0/');
685 
686  foreach($open_search_results as $key => $open_search_result) {
687  $report_root_parameters[$key] = intval($open_search_result);
688  }
689 
690  $google_results = $xml->children('http://schemas.google.com/analytics/2009');
691 
692  foreach($google_results->dataSource->property as $property_attributes) {
693  $attr = $property_attributes->attributes();
694  $report_root_parameters[str_replace('ga:','',$attr->name)] = strval($attr->value);
695  }
696 
697  $report_root_parameters['startDate'] = strval($google_results->startDate);
698  $report_root_parameters['endDate'] = strval($google_results->endDate);
699 
700  //Load result aggregate metrics
701 
702  foreach($google_results->aggregates->metric as $aggregate_metric) {
703  $attr = $aggregate_metric->attributes();
704  $metric_value = strval($attr->value);
705  $name = $attr->name;
706  //Check for float, or value with scientific notation
707  if(preg_match('/^(\d+\.\d+)|(\d+E\d+)|(\d+.\d+E\d+)$/',$metric_value)) {
708  $report_aggregate_metrics[str_replace('ga:','',$name)] = floatval($metric_value);
709  }
710  else {
711  $report_aggregate_metrics[str_replace('ga:','',$name)] = intval($metric_value);
712  }
713  }
714 
715  //Load result entries
716 
717  foreach($xml->entry as $entry) {
718  $metrics = array();
719  $children = $entry->children('http://schemas.google.com/analytics/2009');
720  foreach($children->metric as $metric) {
721  $attr = $metric->attributes();
722  $metric_value = strval($attr->value);
723  $name = $attr->name;
724 
725  //Check for float, or value with scientific notation
726  if(preg_match('/^(\d+\.\d+)|(\d+E\d+)|(\d+.\d+E\d+)$/',$metric_value)) {
727  $metrics[str_replace('ga:','',$name)] = floatval($metric_value);
728  }
729  else {
730  $metrics[str_replace('ga:','',$name)] = intval($metric_value);
731  }
732  }
733 
734  $dimensions = array();
735  $children = $entry->children('http://schemas.google.com/analytics/2009');
736  foreach($children->dimension as $dimension) {
737  $attr = $dimension->attributes();
738  $dimensions[str_replace('ga:','',$attr->name)] = strval($attr->value);
739  }
740 
741  $results[] = array('metrics' => $metrics, 'dimensions' => $dimensions);
742  }
743 
744  return $results;
745  }

747  function generate_google_profiles() {
748 
749  if (!empty($this->ga_token) && empty($this->profiles)) {
750  $profiles = $this->profiles;
751  $profile_options = $this->profile_options;
752  $ga_profile_id = $this->ga_profile_id;
753  $url = 'https://www.googleapis.com/analytics/v2.4/management/accounts/~all/webproperties/~all/profiles';
754 
755  $wp_http = $this->get_wp_http();
756  $request_args = array(
757  'timeout' => 10,
758  'headers' => $this->get_authsub_headers(),
759  'sslverify' => false,
760  'redirections' => 0,
761  );
762  $result = $wp_http->request(
763  $url,
764  $request_args
765  );
766 
767  if (is_wp_error($result)) {
768  $connection_errors = $result->get_error_messages();
769  }
770  else {
771  $http_code = $result['response']['code'];
772 
773  if ($http_code != 200) {
774  $this->ga_auth_error = $result['response']['code'].': '.$result['response']['message'];
775  }
776  else {
777  $xml = simplexml_load_string($result['body']);
778  $profiles = array();
779  foreach($xml->entry as $entry) {
780  $properties = array();
781  $children = $entry->children('http://schemas.google.com/analytics/2009');
782  foreach($children->property as $property) {
783  $attr = $property->attributes();
784  $properties[str_replace('ga:', '', $attr->name)] = strval($attr->value);
785  }
786  $properties['title'] = strval($entry->title);
787  $properties['updated'] = strval($entry->updated);
788  $profiles[$properties['profileId']] = $properties;
789  }
790  if (count($profiles)) {
791  if (empty($this->ga_profile_id)) {
792  $ga_profile_id = $properties['profileId']; // just use the last one
793  update_option('ccStats_ga_profile_id', $ga_profile_id);
794  } else {
795  $ga_profile_id = $this->ga_profile_id;
796  }
797  if (count($profiles) > 1) {
798  $profile_options = array();
799  foreach ($profiles as $id => $profile) {
800  $profile_options[$profile['title']] = '<option value="'.$id.'"'.($ga_profile_id == $id ? 'selected="selected"' : '').'>'.$profile['title'].'</option>';
801  }
802  }
803  }
804  }
805  }
806 
807  $this->profiles = $profiles;
808  ksort($profile_options);
809  $this->profile_options = $profile_options;
810  $this->ga_profile_id = $ga_profile_id;
811  }
812  }
