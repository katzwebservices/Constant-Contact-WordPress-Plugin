98  protected function processForms() {
99 
100  if(!isset($_REQUEST['cc-form-id'])) { return; }
101 
102  global $cc_form_selected_id;
103 
104  // Allowed actions: add, update, delete
105  $action = isset( $_REQUEST['action'] ) ? $_REQUEST['action'] : 'edit';
106 
107  $cc_form_selected_id = cc_form_get_selected_id();
108 
109  if($action === 'edit') { return; }
110 
111  $messages = array();
112 
113  switch ( $action ) {
114  case 'delete_all':
115  delete_option('cc_form_design');
116  break;
117  case 'delete':
118 
119  $cc_form_selected_id = isset($_REQUEST['form']) ? (int)$_REQUEST['form'] : $cc_form_selected_id;
120 
121  if ( $deleted_form = wp_get_cc_form( $cc_form_selected_id ) ) {
122 
123  $delete_cc_form = wp_delete_cc_form( $cc_form_selected_id );
124 
125  if ( is_wp_error($delete_cc_form) ) {
126  $messages[] = '<div id="message" class="error"><p>' . $delete_cc_form->get_error_message() . '</p></div>';
127  } else {
128  $messages[] = '<div id="message" class="updated"><p>' . __('The form '.$deleted_form['form-name'].' has been successfully deleted.','constant-contact-api') . '</p></div>';
129  // Select the next available menu
130  $cc_form_selected_id = -1;
131  $_cc_forms = wp_get_cc_forms( array('orderby' => 'name') );
132  foreach( $_cc_forms as $index => $_cc_form ) {
133  if ( $index == count( $_cc_forms ) - 1 ) {
134  $cc_form_selected_id = $_cc_form['cc-form-id'];
135  break;
136  }
137  }
138  }
139  $_REQUEST['deleted'] = 1;
140  } else {
141  $_REQUEST['deleted'] = 0;
142  // Reset the selected menu
143  $cc_form_selected_id = -1;
144  unset( $_REQUEST['form'] );
145  $messages[] = '<div id="message" class="error"><p>'.__('The form could not be deleted. The form may have already been deleted.', 'constant-contact-api').'</p></div>';
146  }
147  break;
148 
149  case 'update':
150  # check_admin_referer( 'update-nav_menu', 'update-nav-menu-nonce' );
151  // Add Form
152  if ( -1 == $cc_form_selected_id ) {
153  $new_form_title = trim( esc_html( $_REQUEST['form-name'] ) );
154  if($new_form_title == 'Enter form name here') { $new_form_title = ''; }
155 
156  $cc_form_selected_id = wp_create_cc_form();
157 
158  if ( is_wp_error( $cc_form_selected_id ) ) {
159  $messages[] = '<div id="message" class="error"><p>' . $cc_form_selected_id->get_error_message() . '</p></div>';
160  } else {
161  $messages[] = '<div id="message" class="updated"><p>' . sprintf( __('The <strong>%s</strong> form has been successfully created.','constant-contact-api'), $new_form_title ) . '</p></div>';
162  }
163 
164  // update existing form
165  } else {

167  $request = wp_update_cc_form_object($cc_form_selected_id, $_REQUEST);
168  if(!is_wp_error($request)) {
169  $messages[] = '<div id="message" class="updated after-h2 fade"><p>' . sprintf( __('The <strong>%s</strong> form has been updated.','constant-contact-api'), $request['form-name'] ) . '</p></div>';

247 function constant_contact_retrieve_form($formid, $force_update=false, $unique_id = '', $lists = array()) {
248 
249  @$_GET['cc_signup_count']++;
250 
251  $formid = (int)$formid;
252 
253  $success = get_transient($unique_id);
254  $success = !empty($success);
255 
256  $form = get_transient("cc_form_$formid");
257 
258  // If it is an array and we are not forcing an update, return the data
259  if(!empty($form) && !$success && !$force_update && !isset($_GET['asdasdas'])) {
260  do_action('ctct_debug', 'Returning Cached Form #'.$formid, $form);
261 
262  // Basic form validation - make sure it's got basic
263  if(preg_match('/kws_form/m', $form)) {
264  return $form;
265  }
266  }
267 
268  $form = wp_get_cc_form($formid);
269 
270  if($form && is_array($form)) {
271  // Just the items we need, please.
272  unset($form['_wp_http_referer'], $form['update-cc-form-nonce'], $form['meta-box-order-nonce'], $form['closedpostboxesnonce'], $form['save_form'], $form['page'], $form['form-name'], $form['action'], $form['form-style'], $form['presets'], $form['truncated_name']);
273  } else {
274  $form = array('formOnly'=>true);
275  }
276 
277  $form['output'] = 'html';
278  $form['time'] = time();
279  $form['verify'] = sha1($unique_id.$form['time']);
280  $form['echo'] = true;
281  $form['path'] = CC_FORM_GEN_URL;
282  $form['cc_success'] = $success;
283  $form['cc_signup_count'] = $_GET['cc_signup_count'];
284  $form['cc_request'] = empty($_REQUEST['uniqueformid']) ? array() : $_REQUEST;
285  $form['uniqueformid'] = $unique_id;
286  $form['debug'] = (current_user_can('manage_options') && isset($_GET['debug']));
287 
288  // Get the form from form.php
289  $response = wp_remote_request( CC_FORM_GEN_URL.'form.php', array(
290  'method' => 'POST',
291  'timeout' => 20,
292  'body' => $form,
293  'sslverify' => false,
294  'compress' => true,
295  'local' => true
296  ));
297 
298  do_action('ctct_debug', 'Requesting Form Data for Form #'.$formid, array('POST body' => $form, 'POST response:' => $response));
299 
300  if( is_wp_error( $response ) ) {
301  do_action('ctct_log', $response, 'error');
302  if(current_user_can('manage_options')) {
303  echo '<!-- Constant Contact API Error: `wp_remote_post` failed with this error: '.$response->get_error_message().' -->';
304  }
305  return false;
306  } else {
307  $form = $response['body'];
308  if(empty($_POST) && !$success) {
309  // Save the array into the cc_form_id transient with a 30 day expiration
310  $set = set_transient("cc_form_$formid", $form, 60*60*24*30);
311  if(!$set) {
312  do_action('ctct_log', 'Setting cc_form_'.$formid.' Transient failed', $form);
313  }
314  }
315 
316  return $form;
317  }

320 function cc_form_get_selected_id($allForms = array()) {
321 
322  if(isset($_REQUEST['form']) && (empty($_REQUEST['action']) || @$_REQUEST['action'] === 'edit' || @$_REQUEST['action'] === 'update')) {
323  return intval($_REQUEST['form']);
324  }
325 
326 
327  if(empty($allForms)) {
328  $cc_form_selected_id = isset($_REQUEST['cc-form-id']) ? (int)$_REQUEST['cc-form-id'] : -1;
329  } else {
330  $lastForm = end($allForms);
331  $cc_form_selected_id = $lastForm['cc-form-id'];
332 
333  // Intstead of always showing new form, show last form possible.
334  $_REQUEST['form'] = (int)$cc_form_selected_id;
335  $_REQUEST['action'] = 'edit';
336  }
337 
338  return $cc_form_selected_id;
339 }
