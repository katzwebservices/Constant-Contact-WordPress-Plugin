722  /**
723  * Get the id of object, or attempt to convert the argument to an int
724  * @param mixed $item - object or a numeric value
725  * @param string $class_name - class name to test the given object against
726  * @throws IllegalArgumentException - if the item is not an instance of the class name given, or cannot be
727  * converted to a numeric value
728  * @return int
729  */
730  private function getArgumentId($item, $className)
731  {
732  $id = null;
733 
734  if (is_numeric($item)) {
735  $id = $item;
736  } elseif (join('', array_slice(explode('\\', get_class($item)), -1)) == $className) {
737  $id = $item->id;
738  } else {
739  throw new IllegalArgumentException(sprintf(Config::get('errors.id_or_object'), $className));
740  }
741 
742  return $id;
743  }
