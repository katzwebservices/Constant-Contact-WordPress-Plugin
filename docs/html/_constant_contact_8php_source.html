789  /**

791  * @param mixed $item - object or a numeric value
792  * @param string $className - class name to test the given object against
793  * @throws IllegalArgumentException - if the item is not an instance of the class name given, or cannot be
794  * converted to a numeric value
795  * @return int
796  */
797  private function getArgumentId($item, $className)
798  {
799  $id = null;
800 
801  if (is_numeric($item)) {
802  $id = $item;
803  } elseif (join('', array_slice(explode('\\', get_class($item)), -1)) == $className) {
804  $id = $item->id;
805  } else {
806  throw new IllegalArgumentException(sprintf(Config::get('errors.id_or_object'), $className));
807  }
808
