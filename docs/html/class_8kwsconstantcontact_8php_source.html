146  // update the existing contact if address already existed
147  } else {
148

152  // Update the contact details

233  /**
234  * Function to fetch multi-page results. Works for all component types.
235  * @param string $type Component type name
236  * @param int|string $id_or_status Either the ID of the component or the status filter. Depends on the component. Example: `$id_or_status = 'sent';` for `EmailCampaigns` component. Example: `$id_or_status = 13;` for `ContactSends` component.
237  * @param string $param Search filter. Sets the limit for requests.
238  * @param array $results Pass the previous results for recursive calls to the method.
239  * @return array Results array with `id` as key to each key/value pair.
240  */
241  function getAll($type = '', $id_or_status = NULL, $param = NULL, &$results = array()) {
242 
243  do_action('ctct_debug', $type);
244  add_filter('ctct_cache', function() { return 60 * 60 * 48; });
245 
246  if($type === 'EmailCampaigns'){
247  if(!is_null($param)) { $id_or_status = null; }
248  $fetch = $this->{"get{$type}"}(CTCT_ACCESS_TOKEN, $id_or_status, $param);
249  } else if(!is_null($id_or_status)) {
250  if(is_null($param)) { $param = '?limit=500'; }
251  $fetch = $this->{"get{$type}"}(CTCT_ACCESS_TOKEN, $id_or_status, $param);
252  } else {
253  if(is_null($param)) { $param = '?limit=500'; }
254  $fetch = $this->{"get{$type}"}(CTCT_ACCESS_TOKEN, $param);
255  }
256 
257  // If this returns the results directly, not as an object.
258  if(is_array($fetch)) { return $fetch; }
259 
260  foreach($fetch->results as $r) { $results[$r->id] = $r; }
261 
262  if(!empty($fetch->next) && $fetch->next != $param) {
263  $this->getAll($type, $id_or_status, $fetch->next, $results);
264  }
265 
266  return $results;
267  }
