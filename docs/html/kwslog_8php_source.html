141  function debug($message = '') {
142 
143  if(!function_exists('current_user_can') || !current_user_can('manage_options')) { return; }
144 
145  $bt = debug_backtrace();
146 
147  foreach($bt as $call) {
148  if($call['function'] === 'do_action' && $call['args'][0] === self::$slug.'_debug') {
149  $caller = $call;
150  // do_action and title
151  unset($caller['args'][0]);
152  if(is_string($caller['args'][1])) {
153  unset($caller['args'][1]);
154  }
155  break;
156  }
157  if(isset($call['line']) && $call['function'] === 'debug' && $call['class'] === 'KWSLog') {
158  $caller = $call;
159  if(!empty($message)) {
160  unset($caller['args'][0]);
161  }
162  break;
163  }
164  }
165 
166  if(!empty($message) && is_string($message)) {
167  $message = '<h4>'.$message.'</h4>';
168  $hidedata = 'display:none;';
169  } elseif(!is_string($message)){
170  $message = '';
171  $hidedata = '';
172  }
173 
174  $additional_data = '';
175  if(!empty($caller['args'])) {
176  foreach($caller['args'] as &$arg) {
177  if(is_string($arg)){
178  $arg = htmlentities2($arg);
179  }
180  }
181  $additional_data = '<div><a href="#" class="kwslog-toggle">View Data</a></div><div class="data" style="'.$hidedata.'"><pre>'.print_r($caller['args'], true).'</pre></div>';
182  }
183 
184  $debug_message = '<div class="kwslog-debug">'.$message.'<p><code>'.str_replace(ABSPATH, '', $caller['file']).'</code>, line <code>'.$caller['line'].'</code></p>'.$additional_data.'</div>';
185 
186  $logs = (array)$this->logs;
187  $logs[] = $debug_message;
188  $this->logs = $logs;
189 
190  }

204  /**
205  * Create a log message. If the log table doesn't exist (it should be created on activation), create the log table.
206  * @param mixed $message What the log message should include.
207  * @param string $level Debug level. Default: 'debug'
208  * @param boolean $recursive Whether or not the message is being called by itself. Only here to prevent infinite loop.
209  */
210  function log_message($message, $data = '', $level = 'debug', $recursive = false) {
211  global $wpdb;
212 
213  $message = is_string($message) ? $message : print_r($message, true);
214  $data = is_string($data) ? $data : print_r($data, true);
215  $message = $message."<br />".$data;
216 
217  $values = array(
218  'message' => $message,
219  'level' => $level
220  );
221 
222  $wpdb->insert($wpdb->prefix.self::$tablename, $values, array('%s','%d'));
223 
224  if(!empty($wpdb->last_error) && preg_match('/doesn\'t\ exist/ism', $wpdb->last_error) && !$recursive) {
225  $this->activate_plugin();
226  $this->log_message($message, $data, $level, true);
227  }
228  }
