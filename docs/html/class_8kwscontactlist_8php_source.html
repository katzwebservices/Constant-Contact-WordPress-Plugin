139  /**

141  * @param string $as Format of HTML; `list`|`select`
142  * @param array $items List array
143  * @param array $atts Settings; `fill`, `selected`, `format`; `format` should use replacement tags with the tag being the name of the var of the List object you want to replace. For example, `%%name%% (%%contact_count%% Contacts)` will return an item with the content "List Name (140 Contacts)"
144  *
145  * `showhidden` If true, will exclude lists that have a status of "hidden" in http://dotcms.constantcontact.com/docs/contact-list-api/contactlist-collection.html
146  * @return [type] [description]
147  */
148  static function outputHTML($passed_items = array(), $atts = array()) {
149 
150  $settings = wp_parse_args($atts, array(
151  'type' => 'checkboxes',
152  'fill' => true, // Fill data into lists
153  'format' => '<span>%%name%%</span>', // Choose HTML format for each item
154  'id_attr' => 'ctct-%%id%%', // Pass a widget instance
155  'name_attr' => 'lists',
156  'checked' => array(), // If as select, what's active?
157  'include' => array(),
158  'showhidden' => true,
159  ));
160 
161  extract($settings);
162 
163  $items = array();
164 
165  // Tell the cache that if the current requests are forced to be
166  // refreshed, the cache should also reset this key.
167  // See Cache_WP_HTTP
168  add_filter('flush_key', function() { return 'ctct_all_lists'; });
169 
170  if($passed_items === 'all') {
171  $items = WP_CTCT::getInstance()->cc->getAllLists();
172  } elseif(!empty($passed_items) && is_array($passed_items)) {
173  foreach($passed_items as $item) {
174  global $list_id;
175 
176  if($fill) {
177  $list_id = is_object($item) ? $item->id : $item;
178 
179  // Tell Cache_WP_HTTP to use the following key
180  // as the transient name
181  add_filter('ctct_cachekey', function() {
182  global $list_id;
183  return 'ctct_list_'.$list_id;
184  });
185  $item = WP_CTCT::getInstance()->cc->getList(CTCT_ACCESS_TOKEN, $list_id);
186  }
187  $items[] = $item;
188  }
189  }
190 
191  switch($type) {
192  case 'hidden':
193  $before = $before_item = $after_item = $after = '';
194  $format = '<input type="hidden" value="%%id%%" name="%%name_attr%%[]" />';
195  break;
196  case 'ul':
197  $before = '<ul class="ul-square">';
198  $before_item = '<li>';
199  $after_item = '</li>';
200  $after = '</ul>';
201  break;
202  case 'dropdown':
203  case 'select':
204  case 'multiselect':
205  $multiple = '';
206 
207  // Even though the multiselect option is no longer available
208  // in the settings, keep this around for backward compatibility.
209  // And if crazy people want multi-selects
210  if($type === 'select' || $type === 'multiselect') {
211  $multiple = ' multiple="multiple"';
212  }
213  $before = '<select name="%%name_attr%%"'.$multiple.' class="select2 ctct-lists">';
214  $before_item = '<option value="%%id%%">';
215  $after_item = '</option>';
216  $after = '</select>';
217  break;
218  case 'checkbox':
219  case 'checkboxes':
220  $before = '<ul class="ctct-lists ctct-checkboxes">';
221  $before_item = '<li><label for="%%id_attr%%"><input type="checkbox" id="%%id_attr%%" value="%%id%%" name="%%name_attr%%[]" %%checked%% /> ';
222  $after_item = '</label></li>';
223  $after = '</ul>';
224  break;
225  }
226 
227  $output = $before;
228 
229  $items_output = '';
230  foreach($items as &$item) {
231 
232  // If include was specified, then we need to skip lists not included
233  if(is_array($passed_items) && (!empty($include) && !in_array($item->id, $include)) || ($item->status === 'HIDDEN' && !$showhidden)) {

247  $item_output = str_replace('%%contact_count%%', $item->get('contact_count', true), $item_output);
248

251  }
252 
253  $output .= $items_output;
254 
255  $output .= $after;
